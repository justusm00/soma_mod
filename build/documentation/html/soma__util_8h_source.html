153static inline soma_scalar_t calc_bond_length(const soma_scalar_t x_i, const soma_scalar_t x_j, const soma_scalar_t box,
154 const int mic);
155inline soma_scalar_t calc_bond_length(const soma_scalar_t x_i, const soma_scalar_t x_j, const soma_scalar_t box,
156 const int mic)
157{
158 soma_scalar_t r = x_i - x_j + 0 * mic * box; // after "+" to shut up compiler warnings
159#if ( ENABLE_MIC == 1)
160 if (mic)
161 {
162#if ( SINGLE_PRECISION == 1)
163 const soma_scalar_t img = rintf(r / box);
164#else
165 const soma_scalar_t img = rint(r / box);
166#endif //SINGLE_PRECISION
167 r -= img * box;
168 }
169#endif //ENABLE_MIC
170 return r;
171}
