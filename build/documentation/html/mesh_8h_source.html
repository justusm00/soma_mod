43static inline void coord_to_cell_coordinate(const struct Phase *p, const soma_scalar_t rx, const soma_scalar_t ry,
44 const soma_scalar_t rz, int *x, int *y, int *z);
45inline void coord_to_cell_coordinate(const struct Phase *p, const soma_scalar_t rx, const soma_scalar_t ry,
46 const soma_scalar_t rz, int *x, int *y, int *z)
47{
48
49 soma_scalar_t px, py, pz;
50
51 // Fold coordinate back into the box
52 px = rx - p->Lx * (int)(rx * p->iLx);
53 py = ry - p->Ly * (int)(ry * p->iLy);
54 pz = rz - p->Lz * (int)(rz * p->iLz);
55
56 // Assure correct symmetry at coordinate = 0
57 if (px < 0)
58 px = p->Lx + px;
59 if (py < 0)
60 py = p->Ly + py;
61 if (pz < 0)
62 pz = p->Lz + pz;
63
64 // Calculate index
65 *x = (int)(px * p->iLx * p->nx);
66 *y = (int)(py * p->iLy * p->ny);
67 *z = (int)(pz * p->iLz * p->nz);
68}

80#pragma acc routine(cell_coordinate_to_index) seq
81inline uint64_t cell_coordinate_to_index(const struct Phase *p, const int x, const int y, const int z)
82{
83 int xt = x;
84#if ( ENABLE_DOMAIN_DECOMPOSITION == 1 )
85 //For performance reasons compile the domain decomposition only if necessary
86 if (p->args.N_domains_arg > 1)
87 {
88 if (xt >= p->local_nx_high) //Wrap back if necessary
89 xt -= p->nx;
90 if (xt < p->local_nx_low)
91 xt += p->nx;
92 if (xt < p->local_nx_low || xt >= p->local_nx_high)
93 {
94 return UINT64_MAX; //Error, requested index out of local bounds
95 }
96 xt -= p->local_nx_low;
97 }
98#endif //ENABLE_MIC
99 //Unified data layout [type][x][y][z]
100 return xt * p->ny * p->nz + y * p->nz + z;
101}
